void SumDelAB(int A, int B)
{
	int sum=0;
	for(int i=A; i<=B; i++)
	{	
		sum+=SumDel(i);
	}
	cout<<"Ответ:"<<sum<<"\n";
}
int SumDel(int n)
{
    int sum=n+1;
	for(int i=2; i<=n/2; i++)
	{
		if(n % i == 0)
		{
			sum+=i;
		}
	}
	return sum;
}
