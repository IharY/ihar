void SokrD(int A, int B)
{
	if(NOD(A,B)!=1)
	{
	cout<<"Сократимая дробь\n";
	}
	else
	{
    cout<<"Несократимая дробь\n";
    }
}
int NOD(int A, int B)
{
    while(B > 0)
	{
		int buf = A % B;
		A = B;
		B = buf;
	}
	return A;
}
