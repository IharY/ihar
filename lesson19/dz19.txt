int NODA(int A, int B)
{
   int nod;
   if(A > B)
	{
		nod=B;
	}
	else
	{
		nod=A;
	}
	while(A % nod !=0 || B % nod !=0)
	{
		nod--;
	}
	A/nod;
	B/nod;
	return A,B;
}
int NODB(int C, int D)
{
   int nod;
   if(C > D)
	{
		nod=C;
	}
	else

		nod=D;
	}
	while(C % nod !=0 || D % nod !=0)
	{
		nod--;
	}
	C/nod;
	D/nod;
	return C,D;
}
