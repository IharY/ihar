int  Nod(int a, int b);
int  Nok(int a, int b);
int _tmain(int argc, _TCHAR* argv[])
{

 int  Nod(int a, int b)
 {
		int nod;
	if(a > b)
	{
		nod=b;
	}
	else
	{
		nod=a;
	}
	while(a % nod !=0 || b % nod !=0)
	{
		nod--;
	}
    return nod;
 }
int  Nok(int a, int b)
{
	int nok;
	if(a > b)
	{
		nok=b;
	}
	else
	{
		nok=a;
	}
	while( nok % a !=0 || nok % a !=0)
	{
		nok++;
	}
    return nok;
}
